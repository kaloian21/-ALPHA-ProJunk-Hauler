local player = game.Players.LocalPlayer
local PlayerGui = player:WaitForChild("PlayerGui")
local RunService = game:GetService("RunService")

-- GUI Setup
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "ProJunkHaulerGUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = PlayerGui

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 300, 0, 250)
frame.Position = UDim2.new(0.5, -150, 0.3, 0)
frame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
frame.Active = true
frame.Draggable = true
frame.Parent = screenGui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 30)
title.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
title.Text = "Pro Junk Hauler"
title.TextColor3 = Color3.new(1,1,1)
title.TextScaled = true
title.Parent = frame

local minimize = Instance.new("TextButton")
minimize.Size = UDim2.new(0, 30, 0, 30)
minimize.Position = UDim2.new(1, -35, 0, 0)
minimize.Text = "-"
minimize.BackgroundColor3 = Color3.fromRGB(50,50,50)
minimize.TextColor3 = Color3.new(1,1,1)
minimize.Parent = frame

local contents = Instance.new("Frame")
contents.Size = UDim2.new(1, 0, 1, -30)
contents.Position = UDim2.new(0, 0, 0, 30)
contents.BackgroundTransparency = 1
contents.Parent = frame

local function createButton(text, yPos)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1, -20, 0, 40)
    btn.Position = UDim2.new(0, 10, 0, yPos)
    btn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
    btn.TextColor3 = Color3.new(1,1,1)
    btn.TextScaled = true
    btn.Text = text
    btn.Parent = contents
    return btn
end

-- Buttons
local teleportButton = createButton("Teleport to Sell Area", 10)
local dualTeleportButton = createButton("Dual Teleport: OFF", 60)
local fastPickupButton = createButton("Fast Pickup: OFF", 110)

-- Minimize logic
local minimized = false
minimize.MouseButton1Click:Connect(function()
    minimized = not minimized
    if minimized then
        contents.Visible = false
        frame.Size = UDim2.new(0, 300, 0, 30)
        minimize.Text = "+"
    else
        contents.Visible = true
        frame.Size = UDim2.new(0, 300, 0, 250)
        minimize.Text = "-"
    end
end)

-- Teleport to sell area
teleportButton.MouseButton1Click:Connect(function()
    local char = player.Character or player.CharacterAdded:Wait()
    local root = char:WaitForChild("HumanoidRootPart")
    root.CFrame = CFrame.new(550,3,250)
end)

-- Dual teleport logic
local minX, maxX = 694, 775
local minZ, maxZ = 69, 158
local yHeight = 3
local exclMinX, exclMaxX = 708, 762
local exclMinZ, exclMaxZ = 85, 140
local fixedPos = Vector3.new(581, yHeight, 267)
local randomInterval = 1
local fixedInterval = 2
local teleporting = false
local randomTimer = 0
local fixedTimer = 0
local character = player.Character or player.CharacterAdded:Wait()
local hrp = character:WaitForChild("HumanoidRootPart")

local function getRandomPosition()
    while true do
        local x = math.random(minX,maxX)
        local z = math.random(minZ,maxZ)
        if not (x >= exclMinX and x <= exclMaxX and z >= exclMinZ and z <= exclMaxZ) then
            return Vector3.new(x, yHeight, z)
        end
    end
end

dualTeleportButton.MouseButton1Click:Connect(function()
    teleporting = not teleporting
    dualTeleportButton.Text = teleporting and "Dual Teleport: ON" or "Dual Teleport: OFF"
end)

RunService.RenderStepped:Connect(function(dt)
    if teleporting then
        randomTimer = randomTimer + dt
        fixedTimer = fixedTimer + dt
        if randomTimer >= randomInterval then
            hrp.CFrame = CFrame.new(getRandomPosition())
            randomTimer = 0
        end
        if fixedTimer >= fixedInterval then
            hrp.CFrame = CFrame.new(fixedPos)
            fixedTimer = 0
        end
    end
end)

-- Fast pickup toggle
local FAST_HOLD = 0.05
local fastEnabled = false
local originalValues = {}

local function setPromptFast(prompt)
    if prompt and prompt:IsA("ProximityPrompt") then
        if fastEnabled then
            if originalValues[prompt] == nil then
                originalValues[prompt] = prompt.HoldDuration
            end
            pcall(function()
                prompt.HoldDuration = FAST_HOLD
            end)
        else
            if originalValues[prompt] then
                pcall(function()
                    prompt.HoldDuration = originalValues[prompt]
                end)
            end
        end
    end
end

for _, obj in ipairs(workspace:GetDescendants()) do
    setPromptFast(obj)
end
workspace.DescendantAdded:Connect(function(desc)
    setPromptFast(desc)
end)

fastPickupButton.MouseButton1Click:Connect(function()
    fastEnabled = not fastEnabled
    fastPickupButton.Text = fastEnabled and "Fast Pickup: ON" or "Fast Pickup: OFF"
    for _, obj in ipairs(workspace:GetDescendants()) do
        setPromptFast(obj)
    end
end)
